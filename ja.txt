//swapping of 2 numbers with using 3 variable

#include<stdio.h>
int main()
{
  int a=10,b=20,c;
  printf("before swapping A=%d B%d,a,b);
  c=a;
  a=b;
  b=c;
  printf("after the swapping  A=%d B%d,a,b);
}
